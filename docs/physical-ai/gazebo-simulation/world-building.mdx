# World Building in Gazebo: Environments, Objects, and Scenarios

## Introduction to World Building

World building in Gazebo is the process of creating realistic simulation environments that accurately represent the real-world conditions where humanoid robots will operate. A well-designed world includes appropriate physics properties, realistic objects, proper lighting, and environmental conditions that enable effective testing and validation of robot behaviors.

## World File Structure

Gazebo world files are SDF (Simulation Description Format) files that define the complete simulation environment. The basic structure includes physics configuration, lighting, models, and other environmental elements.

### Basic World File Template

```xml
<?xml version="1.0" ?>
<sdf version="1.7">
  <world name="humanoid_test_world">
    <!-- Physics configuration -->
    <physics name="default_physics" type="ode">
      <gravity>0 0 -9.8</gravity>
      <ode>
        <solver>
          <type>quick</type>
          <iters>1000</iters>
          <sor>1.3</sor>
        </solver>
        <constraints>
          <cfm>0.000001</cfm>
          <erp>0.2</erp>
        </constraints>
      </ode>
    </physics>

    <!-- Environment lighting -->
    <light name="sun" type="directional">
      <cast_shadows>true</cast_shadows>
      <pose>0 0 10 0 0 0</pose>
      <diffuse>0.8 0.8 0.8 1</diffuse>
      <specular>0.2 0.2 0.2 1</specular>
      <direction>-0.3 0.3 -1</direction>
    </light>

    <!-- Ground plane -->
    <model name="ground_plane" static="true">
      <link name="link">
        <collision name="collision">
          <geometry>
            <plane>
              <normal>0 0 1</normal>
              <size>100 100</size>
            </plane>
          </geometry>
          <surface>
            <friction>
              <ode>
                <mu>1.0</mu>
                <mu2>1.0</mu2>
              </ode>
            </friction>
          </surface>
        </collision>
        <visual name="visual">
          <geometry>
            <plane>
              <normal>0 0 1</normal>
              <size>100 100</size>
            </plane>
          </geometry>
          <material>
            <ambient>0.7 0.7 0.7 1</ambient>
            <diffuse>0.7 0.7 0.7 1</diffuse>
          </material>
        </visual>
      </link>
    </model>

    <!-- Include robot -->
    <include>
      <uri>model://humanoid_robot</uri>
      <pose>0 0 1.0 0 0 0</pose>
    </include>

    <!-- Additional objects and models -->
    <!-- More content will be added here -->
  </world>
</sdf>
```

## Advanced World Building Techniques

### Using Heightmaps for Complex Terrain

Heightmaps allow you to create realistic outdoor environments with complex topography:

```xml
<model name="complex_terrain" static="true">
  <link name="terrain_link">
    <collision name="collision">
      <geometry>
        <heightmap>
          <!-- URI can point to an image file -->
          <uri>file://meshes/terrain_heightmap.png</uri>
          <size>200 200 50</size>
          <pos>0 0 0</pos>
          <!-- Subsampling for performance -->
          <sampling>4</sampling>
        </heightmap>
      </geometry>
    </collision>
    <visual name="visual">
      <geometry>
        <heightmap>
          <!-- Separate texture for visual appearance -->
          <uri>file://textures/terrain_texture.png</uri>
          <size>200 200 50</size>
          <pos>0 0 0</pos>
          <sampling>4</sampling>
        </heightmap>
      </geometry>
    </visual>
  </link>
</model>
```

### Creating Dynamic Environments

For environments with moving objects:

```xml
<!-- Moving platform -->
<model name="moving_platform" static="false">
  <pose>5 0 0.5 0 0 0</pose>
  <link name="platform_link">
    <inertial>
      <mass>10.0</mass>
      <inertia>
        <ixx>1.0</ixx>
        <ixy>0.0</ixy>
        <ixz>0.0</ixz>
        <iyy>1.0</iyy>
        <iyz>0.0</iyz>
        <izz>1.0</izz>
      </inertia>
    </inertial>
    <collision name="collision">
      <geometry>
        <box>
          <size>2 1 0.2</size>
        </box>
      </geometry>
    </collision>
    <visual name="visual">
      <geometry>
        <box>
          <size>2 1 0.2</size>
        </box>
      </geometry>
      <material>
        <ambient>0.8 0.6 0.2 1</ambient>
        <diffuse>0.8 0.6 0.2 1</diffuse>
      </material>
    </visual>
  </link>
  <!-- Joint to control movement -->
  <joint name="platform_joint" type="prismatic">
    <parent>world</parent>
    <child>platform_link</child>
    <axis>
      <xyz>1 0 0</xyz>
      <limit>
        <lower>-10</lower>
        <upper>10</upper>
        <effort>100</effort>
        <velocity>1.0</velocity>
      </limit>
    </axis>
  </joint>
</model>
```

### Including Models from Model Database

You can include pre-made models from Gazebo's model database:

```xml
<!-- Include a table from the model database -->
<include>
  <uri>model://table</uri>
  <pose>2 0 0 0 0 0</pose>
</include>

<!-- Include a specific model with custom pose -->
<include>
  <uri>model://cylinder</uri>
  <pose>-3 2 0.5 0 0 0</pose>
  <name>cylinder_obstacle</name>
</include>

<!-- Include with custom properties -->
<include>
  <uri>model://ground_plane</uri>
  <pose>0 0 0 0 0 0</pose>
</include>
```

## Performance Optimization for World Building

### Static vs Dynamic Objects

Always mark objects that don't move as static for better performance:

```xml
<!-- Static objects are more efficient -->
<model name="static_wall" static="true">
  <link name="link">
    <collision>
      <geometry>
        <box>
          <size>5 0.2 2</size>
        </box>
      </geometry>
    </collision>
    <visual>
      <geometry>
        <box>
          <size>5 0.2 2</size>
        </box>
      </geometry>
    </visual>
  </link>
</model>

<!-- Only make objects dynamic if they need to move -->
<model name="moving_door" static="false">
  <link name="door_link">
    <inertial>
      <mass>10.0</mass>
      <inertia>
        <ixx>1.0</ixx>
        <ixy>0</ixy>
        <ixz>0</ixz>
        <iyy>1.0</iyy>
        <iyz>0</iyz>
        <izz>1.0</izz>
      </inertia>
    </inertial>
    <!-- Collision and visual elements -->
  </link>
  <joint name="door_hinge" type="revolute">
    <parent>world</parent>
    <child>door_link</child>
    <axis>
      <xyz>0 0 1</xyz>
      <limit>
        <lower>-1.57</lower>
        <upper>1.57</upper>
        <effort>50</effort>
        <velocity>1.0</velocity>
      </limit>
    </axis>
  </joint>
</model>
```

### Simplified Collision Geometry

Use simpler collision geometry for better performance:

```xml
<!-- Use simple collision geometry -->
<collision name="simplified_collision">
  <geometry>
    <!-- Instead of complex mesh, use simple shapes -->
    <box>
      <size>1.0 1.0 1.0</size>
    </box>
  </geometry>
</collision>

<!-- For complex shapes, use multiple simple collision elements -->
<collision name="complex_collision">
  <geometry>
    <!-- Combine multiple simple shapes -->
    <box>
      <size>1.0 0.2 0.2</size>
    </box>
  </geometry>
</collision>
```

## World Building Best Practices

### 1. Consistent Coordinate Systems

Always maintain consistent coordinate systems throughout your world:

```xml
<!-- Use standard coordinate system: X forward, Y left, Z up -->
<!-- Position objects relative to world origin -->
<model name="robot" static="false">
  <pose>0 0 1.0 0 0 0</pose>  <!-- Start at center, 1m above ground -->
</model>
```

### 2. Proper Scaling

Ensure all objects are properly scaled relative to each other:

```xml
<!-- Humanoid robot is typically 1.5-2m tall -->
<!-- Objects should be scaled appropriately -->
<model name="table" static="true">
  <pose>2 0 0 0 0 0</pose>
  <link name="table_link">
    <collision>
      <geometry>
        <box>
          <size>1.0 0.8 0.75</size>  <!-- Standard table height: ~0.75m -->
        </box>
      </geometry>
    </collision>
  </link>
</model>
```

### 3. Realistic Physics Properties

Set appropriate friction and contact properties:

```xml
<collision name="floor_collision">
  <surface>
    <friction>
      <ode>
        <mu>0.8</mu>  <!-- Good friction for humanoid feet -->
        <mu2>0.8</mu2>
      </ode>
    </friction>
  </surface>
</collision>
```

### 4. Modular Design

Organize your world into reusable components:

```xml
<!-- Create separate SDF files for reusable components -->
<!-- Then include them in your main world file -->
<include>
  <uri>model://simple_room</uri>
  <pose>0 0 0 0 0 0</pose>
</include>
```

### Scripted World Generation

```python
#!/usr/bin/env python3
"""
Script to generate complex Gazebo worlds
"""
import xml.etree.ElementTree as ET
from xml.dom import minidom

def create_world_sdf(world_name, models, physics_config):
    """Create a complete world SDF file"""
    sdf = ET.Element("sdf", version="1.7")
    world = ET.SubElement(sdf, "world", name=world_name)

    # Add physics configuration
    physics = ET.SubElement(world, "physics", name="default_physics", type="ode")
    gravity = ET.SubElement(physics, "gravity")
    gravity.text = "0 0 -9.8"

    # Add models to world
    for model in models:
        model_elem = ET.SubElement(world, "model", name=model['name'])
        if 'static' in model:
            model_elem.set('static', str(model['static']).lower())
        if 'pose' in model:
            pose = ET.SubElement(model_elem, "pose")
            pose.text = f"{model['pose'][0]} {model['pose'][1]} {model['pose'][2]} 0 0 0"

        # Add links and other elements based on model definition
        link = ET.SubElement(model_elem, "link", name="link")
        collision = ET.SubElement(link, "collision", name="collision")
        geometry = ET.SubElement(collision, "geometry")
        box = ET.SubElement(geometry, "box")
        size = ET.SubElement(box, "size")
        size.text = f"{model['size'][0]} {model['size'][1]} {model['size'][2]}"

    # Convert to pretty XML string
    rough_string = ET.tostring(sdf, 'unicode')
    reparsed = minidom.parseString(rough_string)
    return reparsed.toprettyxml(indent="  ")

# Example usage
models = [
    {
        'name': 'ground_plane',
        'static': True,
        'pose': [0, 0, 0],
        'size': [10, 10, 0.1]
    },
    {
        'name': 'obstacle',
        'static': True,
        'pose': [2, 0, 0.2],
        'size': [0.5, 0.5, 0.4]
    }
]

world_xml = create_world_sdf("test_world", models, {})
print(world_xml)
```
